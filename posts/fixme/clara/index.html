 1#include <iostream>
 2#include <string>
 3 
 4#include "clara.hpp"
 5 
 6enum class verbosity_level
 7{
 8   low, normal, debug
 9};
10 
11int main(int argc, char const * const * argv)
12{
13   using namespace clara;
14 
15   auto source    = std::string{};
16   auto dest      = std::string{};
17   auto threshold = 0.0;
18   auto ignore    = false;
19   auto verbosity = verbosity_level::low;
20   auto depth     = 0;
21   auto showhelp  = false;
22 
23   auto parser =
24      Opt(threshold, "threshold")["-t"]("Threshold for analysis inclusion") |
25      Opt(ignore)["-i"]["--ignore"]("Ignore conflicts") |
26      Opt(verbosity, "low|normal|debug")["-v"]["--verbosity"]("The verbosity level") |
27      Opt([&depth](int const d) 
28      {
29         if (d < 0 || d > 10)
30         {
31            return ParserResult::runtimeError("Depth must be between 1 and 10");
32         }
33         else
34         {
35            depth = d;
36            return ParserResult::ok(ParseResultType::Matched);
37         }
38      }, "depth")["-d"]("Depth of analysis (1 to 10)") |
39      Arg(source, "source")("The path of the source") |
40      Arg(dest, "destination")("The path to the result") |
41      Help(showhelp);
42 
43   try
44   {
45      auto result = parser.parse(Args(argc, argv));
46      if (!result)
47      {
48         std::cerr << "Error in command line: " << result.errorMessage() << std::endl;
49         return 1;
50      }
51      else if (showhelp)
52      {
53         parser.writeToStream(std::cout);
54      }
55      else
56      {
57         std::cout <<
58            "source:    " << source << std::endl <<
59            "dest:      " << dest << std::endl <<
60            "threshold: " << threshold << std::endl <<
61            "ignore:    " << ignore << std::endl <<
62            "verbosity: " << verbosity << std::endl <<
63            "dept:      " << depth << std::endl;
64      }
65   }
66   catch (std::exception const & e)
67   {
68      std::cout << e.what() << std::endl;
69   }
70 
71   return 0;
72}
