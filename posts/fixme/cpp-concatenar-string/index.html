Concatenar strings

Sumário

  • Rotina de teste
    • Makefile
    • Arquivo de teste
  • Execução do teste

Use o operador += ou a função append sempre que possível.
Adaptado de uma resposta1 do stackoveflow.

  • 13ms: s1 + s2 + s3
  • 5ms: s1 += s2
  • 6ms: s1.append( s2 )
  • 59ms: oss << s1 << s2

Rotina de teste

Arquivo de teste

 1#include <iostream>
 2#include <string>
 3#include <chrono>
 4#include <sstream>
 5
 6int main ()
 7{
 8    typedef std::chrono::high_resolution_clock clock;
 9    typedef std::chrono::duration<float, std::milli> mil;
10    std::string l_czTempStr;
11    std::string s1="Test data1";
12    auto t0 = clock::now();
13    #if VER==1
14    for (int i = 0; i < 100000; ++i)
15    {
16        l_czTempStr = s1 + "Test data2" + "Test data3";
17    }
18    #elif VER==2
19    for (int i = 0; i < 100000; ++i)
20    {
21        l_czTempStr =  "Test data1"; 
22        l_czTempStr += "Test data2";
23        l_czTempStr += "Test data3";
24    }
25    #elif VER==3
26    for (int i = 0; i < 100000; ++i)
27    {
28        l_czTempStr =  "Test data1"; 
29        l_czTempStr.append("Test data2");
30        l_czTempStr.append("Test data3");
31    }
32    #elif VER==4
33    for (int i = 0; i < 100000; ++i)
34    {
35        std::ostringstream oss;
36        oss << "Test data1";
37        oss << "Test data2";
38        oss << "Test data3";
39        l_czTempStr = oss.str();
40    }
41    #endif
42    auto t1 = clock::now();
43    std::cout << l_czTempStr << '\n';
44    std::cout << mil(t1-t0).count() << "ms\n";
45}
