FCGI é um protocolo no qual aplicações conversam com o servidor via requisições web. Foi desenvolvido a muito tempo atrás para mitigar um problema de escalabilidade do CGI. Tipicamente uma aplicação CGI cria um novo processo (fork) para tratar a requisição. Ao mesmo tempo que isto é inconveniente, é terrivelmente ineficiente para lidar com a sobrecarga de criação e finalização de cada processo. A ideia principal do FCGI é lançar (spawn) um certo número de processos estáticos que vão tratar as requisições web. Isto permite lidar melhor com a sobrecarga de criar e terminar processos e reusar os recursos através de várias requisições.

FCGI facilmente permite que sua aplicação aceite requisições web apenas interagindo com a stdio.

As variáveis de ambiente são similares às do CGI (CONTENT_LENGTH, SCRIPT_NAME, …)

Input stream para o corpo da requisição

Output stream para o cabeçalho e corpo da resposta

Output stream para mensagens de log

As partidas e paradas do webserver são independentes das partidas e paradas da aplicação.

Escolha a implementação

  • mod_fcgid:
    • Se voce precisa de algo para gerenciar o ciclo de vida dos processos da aplicação em conjunto com httpd.
    • Monitoramento de processos via relatório do mod_status
    • NÃO tem balanceamento de carga
    • NÃO suporta requisições concorrentes
    • pode trabalhar em par com o mod_authnz_fcgi
    • NÃO suporta múltiplas requisições por conexão
    • suporta várias conexões e sockets simultâneos
  • mod_proxy_fcgi:
    • Se voce precisa da algo diferente ou gerenciamento de processos mais flexível
  • mod_authnz_fcgi:
    • se voce precisa de autorizadores
    • suporta authn e authz
    • Supports the modern provider configuration mechanism
    • Supports application delivery of the response body with a failed authn check
    • Does not share connections with mod proxy fcgi
    • Does not interoperate with mod authn socache
    • Does not support multiple requests per connection
    • Does not support the httpd access control phase

Configuração para detectar processo congelado

By default, if a request does not complete within five minutes the application will be terminated.

# my report generates output over a long period of
# time; don’t kill it
FcgidBusyTimeout 3600

kill anything that doesn’t respond within 30 seconds


FcgidBusyTimeout 30

By default, if no data can be read or written within 40 seconds, the application will be terminated.

# oh, and there are long pauses between generation of any output
FcgidIOTimeout 300

Exemplo bruto de CGI

Processa e sai!

 1int main(int argc, char **argv)
 2{
 3  extern char **environ; /* environ is {"CONTENT_LENGTH=105", "SCRIPT_NAME=/path/to/foo.fcgi", etc. } */
 4  const char *cl_str;
 5  if ((cl_str = getenv("CONTENT_LENGTH")) {
 6    read(FILENO_STDIN,,); /* request body */
 7  }
 8  write(STDOUT_FILENO,,); /* response headers and body */
 9  write(STDERR_FILENO,,); /* to web server log */
10}

Exemplo - echo

  1// http://chriswu.me/code/hello_world_fcgi/main_v2.cpp
  2#include <stdlib.h>
  3#include <iostream>
  4#include <string>
  5#include "fcgio.h"
  6
  7using namespace std;
  8
  9// Maximum bytes
 10const unsigned long STDIN_MAX = 1000000;
 11
 12/**
 13 * Note this is not thread safe due to the static allocation of the
 14 * content_buffer.
 15 */
 16string get_request_content(const FCGX_Request & request) {
 17    char * content_length_str = FCGX_GetParam("CONTENT_LENGTH", request.envp);
 18    unsigned long content_length = STDIN_MAX;
 19
 20    if (content_length_str) {
 21        content_length = strtol(content_length_str, &content_length_str, 10);
 22        if (*content_length_str) {
 23            cerr << "Can't Parse 'CONTENT_LENGTH='"
 24                 << FCGX_GetParam("CONTENT_LENGTH", request.envp)
 25                 << "'. Consuming stdin up to " << STDIN_MAX << endl;
 26        }
 27
 28        if (content_length > STDIN_MAX) {
 29            content_length = STDIN_MAX;
 30        }
 31    } else {
 32        // Do not read from stdin if CONTENT_LENGTH is missing
 33        content_length = 0;
 34    }
 35
 36    char * content_buffer = new char[content_length];
 37    cin.read(content_buffer, content_length);
 38    content_length = cin.gcount();
 39
 40    // Chew up any remaining stdin - this shouldn't be necessary
 41    // but is because mod_fastcgi doesn't handle it correctly.
 42
 43    // ignore() doesn't set the eof bit in some versions of glibc++
 44    // so use gcount() instead of eof()...
 45    do cin.ignore(1024); while (cin.gcount() == 1024);
 46
 47    string content(content_buffer, content_length);
 48    delete [] content_buffer;
 49    return content;
 50}
 51
 52int main(void) {
 53    // Backup the stdio streambufs
 54    streambuf * cin_streambuf  = cin.rdbuf();
 55    streambuf * cout_streambuf = cout.rdbuf();
 56    streambuf * cerr_streambuf = cerr.rdbuf();
 57
 58    FCGX_Request request;
 59
 60    FCGX_Init();
 61    FCGX_InitRequest(&request, 0, 0);
 62
 63    while (FCGX_Accept_r(&request) == 0) {
 64        fcgi_streambuf cin_fcgi_streambuf(request.in);
 65        fcgi_streambuf cout_fcgi_streambuf(request.out);
 66        fcgi_streambuf cerr_fcgi_streambuf(request.err);
 67
 68        cin.rdbuf(&cin_fcgi_streambuf);
 69        cout.rdbuf(&cout_fcgi_streambuf);
 70        cerr.rdbuf(&cerr_fcgi_streambuf);
 71
 72        const char * uri = FCGX_GetParam("REQUEST_URI", request.envp);
 73
 74        string content = get_request_content(request);
 75
 76        if (content.length() == 0) {
 77            content = ", World!";
 78        }
 79
 80        cout << "Content-type: text/html\r\n"
 81             << "\r\n"
 82             << "<html>\n"
 83             << "  <head>\n"
 84             << "    <title>Hello, World!</title>\n"
 85             << "  </head>\n"
 86             << "  <body>\n"
 87             << "    <h1>Hello " << content << " from " << uri << " !</h1>\n"
 88             << "  </body>\n"
 89             << "</html>\n";
 90
 91        // Note: the fcgi_streambuf destructor will auto flush
 92    }
 93
 94    // restore stdio streambufs
 95    cin.rdbuf(cin_streambuf);
 96    cout.rdbuf(cout_streambuf);
 97    cerr.rdbuf(cerr_streambuf);
 98
 99    return 0;
100}

Exemplo em C com libfcgi-dev

 1// apt-get install libfcgi-dev
 2// gcc tiny-fcgi.c -o tiny-fcgi -lfcgi
 3// cp tiny-fcgi /var/www/
 4// sudo /usr/sbin/apachectl restart
 5// http://127.0.0.1/tiny-fcgi
 6
 7#include "fcgi_stdio.h"
 8#include <stdlib.h>
 9
10void main(void)
11{
12    int count = 0;
13    while(FCGI_Accept() >= 0) {
14        printf("Content-type: text/html\r\n"
15               "\r\n"
16               "<title>FastCGI Hello!</title>"
17               "<h1>FastCGI Hello!</h1>"
18               "Request number %d running on host <i>%s</i>\n",
19                ++count, getenv("SERVER_NAME"));
20    }
21}

Configurando o Apache

apt-get install apache2
vi /etc/apache2/sites-enabled/000-default

Exemplo de configuração básica

<Directory /var/www/>
SetHandler fcgid-script
Options +ExecCGI

Customize the next two directives for your requirements.


Order allow,deny
Allow from all
</Directory>

Basta copiar o executável para a pasta /var/www.
