Fontconfig - escondendo fontes

Sumário

  • Meu fonts.conf
  • Referências

Após a instalação do LaTeX muitas fontes matemáticas e algumas muito estranhas ficaram sendo listadas no inkscape, libreoffice, …

É possível especificar quais fontes devem ser listadas pelos programas através do arquivo ~/.config/fontconfig/fonts.conf.

Para obter o caminho das fontes para remoção use fc-list | grep -i "parte do nome da fonte".

 1$ fc-list | grep -i "noto"
 2/home/geraldo/.fonts/Noto/Noto-Mono-for-Powerline.ttf: Noto Mono for Powerline:style=Regular
 3/usr/share/fonts/truetype/noto/NotoSans-Bold.ttf: Noto Sans:style=Bold
 4/usr/share/fonts/truetype/noto/NotoSans-Italic.ttf: Noto Sans:style=Italic
 5/usr/share/fonts/truetype/noto/NotoSerif-Regular.ttf: Noto Serif:style=Regular
 6/usr/share/fonts/truetype/noto/NotoSerif-Italic.ttf: Noto Serif:style=Italic
 7/usr/share/fonts/truetype/noto/NotoMusic-Regular.ttf: Noto Music:style=Regular
 8/usr/share/fonts/truetype/noto/NotoSerif-BoldItalic.ttf: Noto Serif:style=Bold Italic
 9/usr/share/fonts/truetype/noto/NotoSans-BoldItalic.ttf: Noto Sans:style=Bold Italic
10/usr/share/fonts/truetype/noto/NotoMono-Regular.ttf: Noto Mono:style=Regular
11/usr/share/fonts/truetype/noto/NotoSerif-Bold.ttf: Noto Serif:style=Bold
12/usr/share/fonts/truetype/noto/NotoColorEmoji.ttf: Noto Color Emoji:style=Regular
13/usr/share/fonts/truetype/noto/NotoSans-Regular.ttf: Noto Sans:style=Regular

Meu fonts.conf

 1<?xml version='1.0'?>
 2<!DOCTYPE fontconfig SYSTEM 'fonts.dtd'>
 3<!--
 4
 5# Como pesquisar os caminhos das fontes
 6
 7fc-list | less
 8fc-list | grep -i "kacs"
 9fc-match 'Font Name'
10
11# Comando para limpeza do cache
12
13fc-cache -f -v
14
15-->
16<selectfont>
17  <rejectfont>
18    <!-- FONTES QUE NÃO SERÃO LISTADAS MAIS -->
19    <glob>/usr/share/fonts/truetype/*/GFS*</glob>
20    <glob>/usr/share/fonts/opentype/ipafont*</glob>
21    <glob>/usr/share/fonts/truetype/fonts-ukij-uyghur/*</glob>
22    <glob>/usr/share/fonts/opentype/mathjax/*</glob>
23    <glob>/usr/share/fonts/opentype/stix*</glob>
24    <glob>/usr/share/fonts/truetype/arphic/*</glob>
25    <glob>/usr/share/fonts/truetype/fonts-bpg-georgian/*</glob>
26    <glob>/usr/share/fonts/truetype/kacst/*</glob>
27    <glob>/usr/share/fonts/truetype/khmeros/*</glob>
28    <glob>/usr/share/fonts/truetype/lohit-*</glob>
29    <glob>/usr/share/fonts/truetype/lyx/*</glob>
30    <!-- FONTES DO TeX -->
31    <glob>/usr/share/texlive/*</glob>
32    <glob>/usr/share/texmf/*</glob>
33    <!-- REMOVE UM CENTENA DE NOTOS -->
34    <glob>/usr/share/fonts/truetype/noto/NotoKufi*</glob>
35    <glob>/usr/share/fonts/truetype/noto/NotoNaskh*</glob>
36    <glob>/usr/share/fonts/truetype/noto/NotoNastaliq*</glob>
37    <glob>/usr/share/fonts/truetype/noto/NotoSans*</glob>
38    <glob>/usr/share/fonts/truetype/noto/NotoSerif*</glob>
39  </rejectfont>
40  <acceptfont>
41    <!-- PERMITE APENAS ALGUMAS NOTOS FILTRADOS ACIMA -->
42    <glob>/usr/share/fonts/truetype/noto/NotoSans-Regular.ttf</glob>
43    <glob>/usr/share/fonts/truetype/noto/NotoSans-Bold.ttf</glob>
44    <glob>/usr/share/fonts/truetype/noto/NotoSans-BoldItalic.ttf</glob>
45    <glob>/usr/share/fonts/truetype/noto/NotoSans-Italic.ttf</glob>
46    <glob>/usr/share/fonts/truetype/noto/NotoSerif-Regular.ttf</glob>
47    <glob>/usr/share/fonts/truetype/noto/NotoSerif-Bold.ttf</glob>
48    <glob>/usr/share/fonts/truetype/noto/NotoSerif-BoldItalic.ttf</glob>
49    <glob>/usr/share/fonts/truetype/noto/NotoSerif-Italic.ttf</glob>
50    <glob>/usr/share/fonts/noto/NotoSans-*</glob>
51    <glob>/usr/share/fonts/noto/NotoSerif-*</glob>
52  </acceptfont>
53</selectfont>
